user nginx;
worker_processes 1;
pid /var/run/nginx.pid;

events {
    worker_connections 1024;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # Logging
    access_log /var/log/nginx/access.log;
    error_log /var/log/nginx/error.log;

    # Include other configuration files (e.g., for fastcgi)
    include /etc/nginx/fastcgi_params;

    # Your server block should go here inside the http context
    server {
        listen 80;
        server_name localhost;

        root /var/www/html/public;  # Path to Laravel's public directory
        index index.php index.html index.htm;

        # Serve static assets (CSS, JS, images)
        location /assets/ {
            root /var/www/html/public;  # Correct root for assets
            try_files $uri $uri/ =404;
        }

        # Main location for handling all other routes, proxy to PHP-FPM (Laravel app)
        location / {
            try_files $uri $uri/ /index.php?$query_string;
        }

        # PHP-FPM handler for Laravel
        location ~ \.php$ {
            include /etc/nginx/fastcgi_params;
            fastcgi_pass app:9000;  # The app container is running PHP-FPM on port 9000
            fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        }

        # Security: Deny access to sensitive files
        location ~* \.(env|git|svn|htaccess|htpasswd|log|sh)$ {
            deny all;
        }

        # Error handling
        error_page 404 /404.html;
        location = /40x.html { }

        error_page 500 502 503 504 /50x.html;
        location = /50x.html { }
    }
}
